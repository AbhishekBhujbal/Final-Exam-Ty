Q.2 A) Write a Python program to draw scatter plots to compare two features of the iris dataset

import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import pandas as pd
from sklearn.datasets import load_iris
In[2]:
iris = sns.load_dataset('iris')

sns.scatterplot(data=iris, x='sepal_length', y='sepal_width', hue='species')
plt.title('Scatter Plot: Sepal Length vs Sepal Width')
plt.xlabel('Sepal Length (cm)')
plt.ylabel('Sepal Width (cm)')

plt.show()

Q.2 B)Create two lists, one representing subject names and the other representing marks obtained in those subjects. Display the data in a pie chart and bar chart.

In[4]:
import matplotlib.pyplot as plt

subjects = ['Math', 'Physics', 'Chemistry', 'English', 'Biology']
marks = [85, 90, 78, 88, 92]


plt.subplot(1, 2, 1) 
plt.pie(marks, labels=subjects, autopct='%1.1f%%')
plt.title('Pie Chart of Marks')

plt.subplot(1, 2, 2)  
plt.bar(subjects, marks, color='skyblue')
plt.xlabel('Subjects')
plt.ylabel('Marks')
plt.title('Bar Chart of Marks')

plt.show()

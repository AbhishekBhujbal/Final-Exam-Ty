data = np.random.randint(1, 101, size=50)

plt.subplot(2, 2, 1)
plt.plot(data, marker='*', color='yellow')
plt.title('Line Plot')
plt.xlabel('Index')
plt.ylabel('Value')

plt.subplot(2, 2, 2)
plt.scatter(range(len(data)), data, color='r')
plt.title('Scatter Plot')
plt.xlabel('Index')
plt.ylabel('Value')

plt.subplot(2, 2, 3)
plt.hist(data, bins=10, color='red')
plt.title('Histogram')
plt.xlabel('Value')
plt.ylabel('Frequency')


plt.subplot(2, 2, 4)
sns.boxplot(data=data, color='cyan')
plt.title('Box Plot')

plt.show()
